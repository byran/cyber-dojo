def save_files(delta, files)
  delta[:deleted].each { |filename| git_rm(filename) }
  delta[:new    ].each { |filename| git_add(filename, filter(filename, files)) }
  delta[:changed].each { |filename|   write(filename, filter(filename, files)) }
end
